** start of script.js **

// Fetch weather data from the proxy API
const getWeather = async (city) => {
  try {
    const response = await fetch(`https://weather-proxy.freecodecamp.rocks/api/city/${city}`);
    
    // Check if the response is actually okay before parsing JSON
    if (!response.ok) {
      throw new Error(`Error fetching data: ${response.statusText}`);
    }
    
    const data = await response.json();
    return data;
  } catch (error) {
    console.error(error);
    // Returning undefined allows showWeather to handle the error state
    return undefined; 
  }
};

// Display weather data on the page
const showWeather = async (city) => {
  const data = await getWeather(city);

  // Handle the error/undefined case (Test 17 & 24)
  if (!data) {
    alert("Something went wrong, please try again later.");
    return;
  }

  // Helper function to handle undefined data by replacing it with "N/A"
  const getSafeValue = (val) => (val !== undefined ? val : "N/A");

  // Populate HTML elements with exact required IDs
  document.getElementById("location").textContent = getSafeValue(data.name);
  document.getElementById("weather-main").textContent = getSafeValue(data.weather?.[0]?.main);
  document.getElementById("main-temperature").textContent = getSafeValue(data.main?.temp);
  document.getElementById("feels-like").textContent = getSafeValue(data.main?.feels_like);
  document.getElementById("humidity").textContent = getSafeValue(data.main?.humidity);
  document.getElementById("wind").textContent = getSafeValue(data.wind?.speed);
  document.getElementById("wind-gust").textContent = getSafeValue(data.wind?.gust);

  // Update weather icon
  const iconElement = document.getElementById("weather-icon");
  const iconSrc = data.weather?.[0]?.icon;
  if (iconSrc) {
    iconElement.src = iconSrc;
    iconElement.style.display = "block";
  } else {
    iconElement.src = "";
    iconElement.style.display = "none";
  }
};

// Event listener for the button
const getBtn = document.getElementById("get-weather-btn");
const citySelect = document.getElementById("city-select");

getBtn.addEventListener("click", () => {
  const selectedCity = citySelect.value;
  // If no city is selected, do nothing
  if (selectedCity) {
    showWeather(selectedCity);
  }
});
